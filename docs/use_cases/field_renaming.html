Write migration Script¶

in the pre-migration.py script add:

from openupgradelib import openupgrade

def _rename_fields(env):
    openupgrade.rename_fields(
        env,
        [
            (
                "mail.activity.type",
                "mail_activity_type",
                "default_description",
                "default_note",
            ),
        ]
    )

@openupgrade.migrate()
def migrate(env, version):
    _rename_fields(env)
